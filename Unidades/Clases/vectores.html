package EJERCICIOS;
import java.util.Scanner;
public class ClaseMatriz {
static int [][] matriz1 = null; // para usar dentro del main o dentro de otros métodos
// null porque necesito que el usuario defina un numero
static Scanner sc = new Scanner(System.in);

public static void main(String[] args) {
// TODO Auto-generated method stub
matriz1 = DefinirTamano(matriz1);
System.out.println("Filas: "+ matriz1.length);
System.out.println("Columnas: "+ matriz1[0].length);

System.out.println(" ");

IngresarValor(matriz1);
System.out.println(" ");
ImprimirMatriz(matriz1);
}
public static int [][]DefinirTamano(int [][] x){
System.out.println("Igrese el tamaño de la matriz(filas&Columnas): ");
String datosIngresados = sc.next();//Captura los datos ingresados
String [] datos = datosIngresados.split("&");//Para que el vector se almacene en 2 elementos(filas y columnas)

int filas = Integer.parseInt(datos[0]);//Convertir de cadena a entero
int columnas = Integer.parseInt(datos[1]);
x = new int [filas][columnas];
return x;
}

public static void IngresarValor(int [][] x ) {
System.out.println("Valores de la matriz: ");
for(int i=0; i< x.length; i++) {
for(int j=0; j< x.length; j++) {//hasta el contenido de la matriz
x[i][j] = Integer.parseInt(sc.next()); // convertir a entero

}

}
}
public static void ImprimirMatriz(int [][]x) {
for (int i= 0 ; i< x.length;i++) {
System.out.println(" ");
for (int j= 0 ; j< x[i].length;j++) {
System.out.print(" V["+x[i][j]+"] ");
}
System.out.println(" ");
}
}
